1use std::time::Instant;

7/// Benchmark configuration

10struct Args {
11    /// Database path
12    #[arg(short, long, default_value = "./benchmark_db")]
13    db_path: String,
14    /// Number of operations
15    #[arg(short, long, default_value_t = 100000)]
16    operations: usize,
17    /// Key size in bytes
18    #[arg(long, default_value_t = 16)]
19    key_size: usize,
20    /// Value size in bytes
21    #[arg(long, default_value_t = 1024)]
22    value_size: usize,
23    /// Workload type (random, sequential)
24    #[arg(long, default_value = "random")]
25    workload: String,
26    /// Whether to test large-value separation
27    #[arg(long, default_value_t = false)]
28    large_values: bool,
29    /// Batch size for batch operations
30    #[arg(long, default_value_t = 100)]
31    batch_size: usize,
32}

34fn random_key(size: usize) -> Vec<u8> {
35    (0..size).map(|_| rand::random::<u8>()).collect()
36}
37
38fn random_value(size: usize) -> Vec<u8> {
39    (0..size).map(|_| rand::random::<u8>()).collect()
40}
