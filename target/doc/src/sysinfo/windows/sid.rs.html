5use windows::core::{PCWSTR, PWSTR};
6use windows::Win32::Foundation::{LocalFree, ERROR_INSUFFICIENT_BUFFER, HLOCAL, PSID};
7use windows::Win32::Security::Authorization::{ConvertSidToStringSidW, ConvertStringSidToSidW};
8use windows::Win32::Security::{

12use crate::sys::utils::to_str;

14#[doc = include_str!("../../md_doc/sid.md")]
15#[derive(Clone, Debug, PartialEq, Eq, PartialOrd, Ord, Hash)]
16pub struct Sid {
17    sid: Vec<u8>,
18}
19
20impl Sid {
21    /// Creates an `Sid` by making a copy of the given raw SID.
22    pub(crate) unsafe fn from_psid(psid: PSID) -> Option<Self> {
23        if psid.is_invalid() {
24            return None;
25        }
26
27        if !IsValidSid(psid).as_bool() {
28            return None;
29        }
30
31        let length = GetLengthSid(psid);
32
33        let mut sid = vec![0; length as usize];
34
35        if CopySid(length, PSID(sid.as_mut_ptr().cast()), psid).is_err() {
36            sysinfo_debug!("CopySid failed: {:?}", std::io::Error::last_os_error());
37            return None;
38        }
39
40        // We are making assumptions about the SID internal structure,
41        // and these only hold if the revision is 1
42        // https://learn.microsoft.com/en-us/windows/win32/api/winnt/ns-winnt-sid
43        // Namely:
44        // 1. SIDs can be compared directly (memcmp).
45        // 2. Following from this, to hash a SID we can just hash its bytes.
46        // These are the basis for deriving PartialEq, Eq, and Hash.
47        // And since we also need PartialOrd and Ord, we might as well derive them
48        // too. The default implementation will be consistent with Eq,
49        // and we don't care about the actual order, just that there is one.
50        // So it should all work out.
51        // Why bother with this? Because it makes the implementation that
52        // much simpler :)
53        assert_eq!(sid[0], 1, "Expected SID revision to be 1");
54
55        Some(Self { sid })
56    }
57
58    /// Retrieves the account name of this SID.
59    pub(crate) fn account_name(&self) -> Option<String> {
60        unsafe {
61            let mut name_len = 0;
62            let mut domain_len = 0;
63            let mut name_use = SidTypeUnknown;
64
65            let sid = PSID((self.sid.as_ptr() as *mut u8).cast());
66            if let Err(err) = LookupAccountSidW(
67                PCWSTR::null(),
68                sid,
69                PWSTR::null(),
70                &mut name_len,
71                PWSTR::null(),
72                &mut domain_len,
73                &mut name_use,
74            ) {
75                if err.code() != ERROR_INSUFFICIENT_BUFFER.to_hresult() {
76                    sysinfo_debug!("LookupAccountSidW failed: {:?}", err);
77                    return None;
78                }
79            }
80
81            let mut name = vec![0; name_len as usize];
82
83            // Reset length to 0 since we're still passing a NULL pointer
84            // for the domain.
85            domain_len = 0;
86
87            if LookupAccountSidW(
88                PCWSTR::null(),
89                sid,
90                PWSTR::from_raw(name.as_mut_ptr()),
91                &mut name_len,
92                PWSTR::null(),
93                &mut domain_len,
94                &mut name_use,
95            )
96            .is_err()
97            {
98                sysinfo_debug!(
99                    "LookupAccountSidW failed: {:?}",
100                    std::io::Error::last_os_error()
101                );
102                return None;
103            }
104
105            Some(to_str(PWSTR::from_raw(name.as_mut_ptr())))
106        }
107    }
108}
109
110impl Display for Sid {
111    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
112        unsafe fn convert_sid_to_string_sid(sid: PSID) -> Option<String> {
113            let mut string_sid = PWSTR::null();
114            if let Err(_err) = ConvertSidToStringSidW(sid, &mut string_sid) {
115                sysinfo_debug!("ConvertSidToStringSidW failed: {:?}", _err);
116                return None;
117            }
118            let result = to_str(string_sid);
119            let _err = LocalFree(HLOCAL(string_sid.0 as _));
120            Some(result)
121        }
122
123        let string_sid =
124            unsafe { convert_sid_to_string_sid(PSID((self.sid.as_ptr() as *mut u8).cast())) };
125        let string_sid = string_sid.ok_or(std::fmt::Error)?;
126
127        write!(f, "{string_sid}")
128    }
129}
130
131impl FromStr for Sid {
132    type Err = String;
133
134    fn from_str(s: &str) -> Result<Self, Self::Err> {
135        unsafe {
136            let mut string_sid: Vec<u16> = s.encode_utf16().collect();
137            string_sid.push(0);
138
139            let mut psid = PSID::default();
140            if let Err(err) =
141                ConvertStringSidToSidW(PCWSTR::from_raw(string_sid.as_ptr()), &mut psid)
142            {
143                return Err(format!("ConvertStringSidToSidW failed: {:?}", err));
144            }
145            let sid = Self::from_psid(psid);
146            let _err = LocalFree(HLOCAL(psid.0 as _));
147
148            // Unwrapping because ConvertStringSidToSidW should've performed
149            // all the necessary validations. If it returned an invalid SID,
150            // we better fail fast.
151            Ok(sid.unwrap())
152        }
153    }
154}
