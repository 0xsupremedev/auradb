3use std::ptr::null_mut;
4
5use windows::Win32::Foundation::{ERROR_BUFFER_OVERFLOW, ERROR_SUCCESS};
6use windows::Win32::NetworkManagement::IpHelper::{
7    GetAdaptersAddresses, GAA_FLAG_SKIP_ANYCAST, GAA_FLAG_SKIP_DNS_SERVER, GAA_FLAG_SKIP_MULTICAST,
8    IP_ADAPTER_ADDRESSES_LH,
9};
10use windows::Win32::Networking::WinSock::AF_UNSPEC;
11
12use crate::common::MacAddr;
13
14/// this iterator yields an interface name and address
15pub(crate) struct InterfaceAddressIterator {
16    /// The first item in the linked list
17    buf: *mut IP_ADAPTER_ADDRESSES_LH,
18    /// The current adapter
19    adapter: *mut IP_ADAPTER_ADDRESSES_LH,
20}
21
22impl InterfaceAddressIterator {
23    fn new() -> Self {
24        Self {
25            buf: null_mut(),
26            adapter: null_mut(),
27        }
28    }
29    unsafe fn realloc(mut self, size: libc::size_t) -> Result<Self, String> {
30        let new_buf = libc::realloc(self.buf as _, size) as *mut IP_ADAPTER_ADDRESSES_LH;
31        if new_buf.is_null() {
32            // insufficient memory available
33            // https://learn.microsoft.com/en-us/cpp/c-runtime-library/reference/malloc?view=msvc-170#return-value
34            // malloc is not documented to set the last-error code
35            Err("failed to allocate memory for IP_ADAPTER_ADDRESSES".to_string())
36        } else {
37            self.buf = new_buf;
38            self.adapter = new_buf;
39            Ok(self)
40        }
41    }
42}
43
44impl Iterator for InterfaceAddressIterator {
45    type Item = (String, MacAddr);
46
47    fn next(&mut self) -> Option<Self::Item> {
48        if self.adapter.is_null() {
49            return None;
50        }
51        unsafe {
52            let adapter = self.adapter;
53            // Move to the next adapter
54            self.adapter = (*adapter).Next;
55            if let Ok(interface_name) = (*adapter).FriendlyName.to_string() {
56                // take the first 6 bytes and return the MAC address instead
57                let [mac @ .., _, _] = (*adapter).PhysicalAddress;
58                Some((interface_name, MacAddr(mac)))
59            } else {
60                // Not sure whether error can occur when parsing adapter name.
61                self.next()
62            }
63        }
64    }
65}
66
67impl Drop for InterfaceAddressIterator {
68    fn drop(&mut self) {
69        unsafe {
70            libc::free(self.buf as _);
71        }
72    }
73}
74
75pub(crate) fn get_interface_address() -> Result<InterfaceAddressIterator, String> {
76    unsafe {
77        // https://learn.microsoft.com/en-us/windows/win32/api/iphlpapi/nf-iphlpapi-getadaptersaddresses#remarks
78        // A 15k buffer is recommended
79        let mut size: u32 = 15 * 1024;
80        let mut ret = ERROR_SUCCESS.0;
81        let mut iterator = InterfaceAddressIterator::new();
82
83        // https://learn.microsoft.com/en-us/windows/win32/api/iphlpapi/nf-iphlpapi-getadaptersaddresses#examples
84        // Try to retrieve adapter information up to 3 times
85        for _ in 0..3 {
86            iterator = iterator.realloc(size as _)?;
87            ret = GetAdaptersAddresses(
88                AF_UNSPEC.0.into(),
89                GAA_FLAG_SKIP_MULTICAST | GAA_FLAG_SKIP_ANYCAST | GAA_FLAG_SKIP_DNS_SERVER,
90                None,
91                Some(iterator.buf),
92                &mut size,
93            );
94            if ret == ERROR_SUCCESS.0 {
95                return Ok(iterator);
96            } else if ret != ERROR_BUFFER_OVERFLOW.0 {
97                break;
98            }
99            // if the given memory size is too small to hold the adapter information,
100            // the SizePointer returned will point to the required size of the buffer,
101            // and we should continue.
102            // Otherwise, break the loop and check the return code again
103        }
104
105        Err(format!("GetAdaptersAddresses() failed with code {ret}"))
106    }
107}
