20pub(crate) struct ComponentInner {
21    temperature: f32,
22    max: f32,
23    critical: Option<f32>,
24    label: String,
25    connection: Option<Connection>,
26}
27
28impl ComponentInner {
29    /// Creates a new `ComponentInner` with the given information.
30    fn new() -> Option<Self> {
31        let mut c = Connection::new()
32            .and_then(|x| x.initialize_security())
33            .and_then(|x| x.create_instance())
34            .and_then(|x| x.connect_server())
35            .and_then(|x| x.set_proxy_blanket())
36            .and_then(|x| x.exec_query())?;
37
38        c.temperature(true)
39            .map(|(temperature, critical)| ComponentInner {
40                temperature,
41                label: "Computer".to_owned(),
42                max: temperature,
43                critical,
44                connection: Some(c),
45            })
46    }
47
48    pub(crate) fn temperature(&self) -> f32 {
49        self.temperature
50    }
51
52    pub(crate) fn max(&self) -> f32 {
53        self.max
54    }
55
56    pub(crate) fn critical(&self) -> Option<f32> {
57        self.critical
58    }
59
60    pub(crate) fn label(&self) -> &str {
61        &self.label
62    }
63
64    pub(crate) fn refresh(&mut self) {
65        if self.connection.is_none() {
66            self.connection = Connection::new()
67                .and_then(|x| x.initialize_security())
68                .and_then(|x| x.create_instance())
69                .and_then(|x| x.connect_server())
70                .and_then(|x| x.set_proxy_blanket());
71        }
72        self.connection = if let Some(x) = self.connection.take() {
73            x.exec_query()
74        } else {
75            None
76        };
77        if let Some(ref mut connection) = self.connection {
78            if let Some((temperature, _)) = connection.temperature(false) {
79                self.temperature = temperature;
80                if self.temperature > self.max {
81                    self.max = self.temperature;
82                }
83            }
84        }
85    }
86}

123macro_rules! bstr {
124    ($x:literal) => {{
125        SysAllocString(w!($x))
126    }};
127}

129struct Connection {
130    instance: Option<IWbemLocator>,
131    server_connection: Option<IWbemServices>,
132    enumerator: Option<IEnumWbemClassObject>,
133    initialized: bool,
134}
135
136#[allow(clippy::non_send_fields_in_send_ty)]
137unsafe impl Send for Connection {}
138unsafe impl Sync for Connection {}
139
140impl Connection {
141    #[allow(clippy::unnecessary_wraps)]
142    fn new() -> Option<Connection> {
143        let val = unsafe { CoInitializeEx(None, Default::default()) };
144        Some(Connection {
145            instance: None,
146            server_connection: None,
147            enumerator: None,
148            initialized: val.is_ok(),
149        })
150    }
151
152    fn initialize_security(self) -> Option<Connection> {
153        unsafe {
154            CoInitializeSecurity(
155                PSECURITY_DESCRIPTOR::default(),
156                -1,
157                None,
158                None,
159                RPC_C_AUTHN_LEVEL_DEFAULT,
160                RPC_C_IMP_LEVEL_IMPERSONATE,
161                None,
162                EOAC_NONE,
163                None,
164            )
165        }
166        .map_or(None, |_| Some(self))
167    }
168
169    fn create_instance(mut self) -> Option<Connection> {
170        let instance =
171            unsafe { CoCreateInstance(&WbemLocator, None, CLSCTX_INPROC_SERVER) }.ok()?;
172        self.instance = Some(instance);
173        Some(self)
174    }
175
176    fn connect_server(mut self) -> Option<Connection> {
177        let instance = self.instance.as_ref()?;
178        let svc = unsafe {
179            let s = bstr!("root\\WMI");
180            let res = instance.ConnectServer(
181                &s,
182                &Default::default(),
183                &Default::default(),
184                &Default::default(),
185                0,
186                &Default::default(),
187                None,
188            );
189            SysFreeString(&s);
190            res
191        }
192        .ok()?;
193
194        self.server_connection = Some(svc);
195        Some(self)
196    }
197
198    fn set_proxy_blanket(self) -> Option<Connection> {
199        unsafe {
200            CoSetProxyBlanket(
201                self.server_connection.as_ref()?,
202                RPC_C_AUTHN_WINNT,
203                RPC_C_AUTHZ_NONE,
204                None,
205                RPC_C_AUTHN_LEVEL_CALL,
206                RPC_C_IMP_LEVEL_IMPERSONATE,
207                None,
208                EOAC_NONE,
209            )
210        }
211        .ok()?;
212
213        Some(self)
214    }
215
216    fn exec_query(mut self) -> Option<Connection> {
217        let server_connection = self.server_connection.as_ref()?;
218
219        let enumerator = unsafe {
220            let s = bstr!("WQL"); // query kind
221            let query = bstr!("SELECT * FROM MSAcpi_ThermalZoneTemperature");
222            let hres = server_connection.ExecQuery(
223                &s,
224                &query,
225                WBEM_FLAG_FORWARD_ONLY | WBEM_FLAG_RETURN_IMMEDIATELY,
226                None,
227            );
228            SysFreeString(&s);
229            SysFreeString(&query);
230            hres
231        }
232        .ok()?;
233
234        self.enumerator = Some(enumerator);
235        Some(self)
236    }
237
238    fn temperature(&mut self, get_critical: bool) -> Option<(f32, Option<f32>)> {
239        let enumerator = self.enumerator.take()?;
240
241        let mut nb_returned = 0;
242        let mut obj = [None; 1];
243
244        unsafe {
245            let _r = enumerator.Next(
246                WBEM_INFINITE, // Time out
247                obj.as_mut_slice(),
248                &mut nb_returned,
249            );
250
251            if nb_returned == 0 {
252                return None; // not enough rights I suppose...
253            }
254
255            let class_obj = match &mut obj {
256                [Some(co)] => co,
257                _ => return None,
258            };
259
260            let _r = class_obj.BeginEnumeration(WBEM_FLAG_NONSYSTEM_ONLY.0);
261
262            let mut variant = std::mem::MaybeUninit::<VARIANT>::uninit();
263            // `Get` only initializes the variant if it succeeds, early returning is not a problem
264            //
265            // <https://learn.microsoft.com/en-us/windows/win32/api/wbemcli/nf-wbemcli-iwbemclassobject-get>
266            class_obj
267                .Get(
268                    w!("CurrentTemperature"),
269                    0,
270                    variant.as_mut_ptr(),
271                    None,
272                    None,
273                )
274                .ok()?;
275
276            let mut variant = variant.assume_init();
277
278            // temperature is given in tenth of degrees Kelvin
279            let temp = (variant.Anonymous.decVal.Anonymous2.Lo64 / 10) as f32 - 273.15;
280            let _r = VariantClear(&mut variant);
281
282            let mut critical = None;
283            if get_critical {
284                class_obj
285                    .Get(w!("CriticalTripPoint"), 0, &mut variant, None, None)
286                    .ok()?;
287
288                // temperature is given in tenth of degrees Kelvin
289                critical = Some((variant.Anonymous.decVal.Anonymous2.Lo64 / 10) as f32 - 273.15);
290                let _r = VariantClear(&mut variant);
291            }
292
293            Some((temp, critical))
294        }
295    }
296}

298impl Drop for Connection {
299    fn drop(&mut self) {
300        // Those three calls are here to enforce that they get dropped in the good order.
301        self.enumerator.take();
302        self.server_connection.take();
303        self.instance.take();
304        if self.initialized {
305            unsafe { CoUninitialize() };
306        }
307    }
