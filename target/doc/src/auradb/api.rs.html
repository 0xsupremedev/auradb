10pub trait Engine: Send + Sync {

14    /// Get a value by key
15    async fn get(&self, key: &Key) -> Result<Option<Value>>;

17    /// Delete a key

20    /// Scan a range of keys

23    /// Write a batch of operations

64/// Main AuraDB engine implementation
65pub struct AuraEngine {
66    /// Engine configuration
67    config: Config,
68    /// In-memory storage (simplified for now)
69    storage: Arc<RwLock<HashMap<Vec<u8>, Vec<u8>>>>,
70    /// Engine status
71    closed: Arc<RwLock<bool>>,
72}
73
74impl AuraEngine {
75    /// Create a new engine instance
76    pub fn new(config: Config) -> Result<Self> {
77        // Create directories
78        std::fs::create_dir_all(&config.db_path)
79            .map_err(|e| crate::error::Error::Io(e))?;
80        
81        // Create WAL and value log directories if they don't exist
82        std::fs::create_dir_all(&config.wal.wal_path)
83            .map_err(|e| crate::error::Error::Io(e))?;
84        std::fs::create_dir_all(&config.value_log.vlog_path)
85            .map_err(|e| crate::error::Error::Io(e))?;
86        
87        Ok(Self {
88            config,
89            storage: Arc::new(RwLock::new(HashMap::new())),
90            closed: Arc::new(RwLock::new(false)),
91        })
92    }
93    
94    /// Put a string key-value pair (convenience method)
95    pub fn put_str(&self, key: &str, value: &str) -> Result<()> {
96        let key = Key::new(key.as_bytes().to_vec());
97        let value = Value::new(value.as_bytes().to_vec());
98        
99        let mut storage = self.storage.write();
100        storage.insert(key.data, value.data);
101        Ok(())
102    }
103    
104    /// Get a string value by key (convenience method)
105    pub fn get_str(&self, key: &str) -> Result<Option<String>> {
106        let key = Key::new(key.as_bytes().to_vec());
107        
108        let storage = self.storage.read();
109        if let Some(value_data) = storage.get(&key.data) {
110            Ok(Some(String::from_utf8_lossy(value_data).to_string()))
111        } else {
112            Ok(None)
113        }
114    }
115    
116    /// Delete a string key (convenience method)
117    pub fn delete_str(&self, key: &str) -> Result<()> {
118        let key = Key::new(key.as_bytes().to_vec());
119        
120        let mut storage = self.storage.write();
121        storage.remove(&key.data);
122        Ok(())
123    }
124    
125    /// Scan string keys in a range (convenience method)
126    pub fn scan_str(&self, start: &str, end: &str) -> Result<Vec<(String, String)>> {
127        let start_key = Key::new(start.as_bytes().to_vec());
128        let end_key = Key::new(end.as_bytes().to_vec());
129        
130        let storage = self.storage.read();
131        let mut results = Vec::new();
132        
133        for (key_data, value_data) in storage.iter() {
134            if key_data >= &start_key.data && key_data <= &end_key.data {
135                let key = String::from_utf8_lossy(key_data).to_string();
136                let value = String::from_utf8_lossy(value_data).to_string();
137                results.push((key, value));
138            }
139        }
140        
141        Ok(results)
142    }
143    
144    /// Write a batch of key-value pairs
145    pub fn write_batch(&self, batch: &[(Vec<u8>, Vec<u8>)]) -> Result<()> {
146        let mut storage = self.storage.write();
147        for (key, value) in batch {
148            storage.insert(key.clone(), value.clone());
149        }
150        Ok(())
151    }
152
153    /// Put a key-value pair using Vec<u8> (for benchmarks)
154    pub fn put_bytes(&self, key: &[u8], value: &[u8]) -> Result<()> {
155        let mut storage = self.storage.write();
156        storage.insert(key.to_vec(), value.to_vec());
157        Ok(())
158    }
159
160    /// Get a value by key using Vec<u8> (for benchmarks)
161    pub fn get_bytes(&self, key: &[u8]) -> Result<Option<Vec<u8>>> {
162        let storage = self.storage.read();
163        Ok(storage.get(key).cloned())
164    }
165}

167#[async_trait::async_trait]
168impl Engine for AuraEngine {
169    async fn put(&self, key: Key, value: Value) -> Result<()> {
170        let mut storage = self.storage.write();
171        storage.insert(key.data, value.data);
172        Ok(())
173    }
174    
175    async fn get(&self, key: &Key) -> Result<Option<Value>> {
176        let storage = self.storage.read();
177        if let Some(value_data) = storage.get(&key.data) {
178            Ok(Some(Value::new(value_data.clone())))
179        } else {
180            Ok(None)
181        }
182    }
183    
184    async fn delete(&self, key: &Key) -> Result<()> {
185        let mut storage = self.storage.write();
186        storage.remove(&key.data);
187        Ok(())
188    }
189    
190    async fn scan(&self, range: Range) -> Result<Vec<(Key, Value)>> {
191        let storage = self.storage.read();
192        let mut results = Vec::new();
193        
194        for (key_data, value_data) in storage.iter() {
195            if key_data >= &range.start.data && key_data <= &range.end.data {
196                let key = Key::new(key_data.clone());
197                let value = Value::new(value_data.clone());
198                results.push((key, value));
199            }
200        }
201        
202        Ok(results)
203    }
204    
205    async fn write_batch(&self, batch: &Batch) -> Result<()> {
206        let mut storage = self.storage.write();
207        
208        for entry in &batch.operations {
209            match entry.op_type {
210                crate::storage::OpType::Put => {
211                    if let Some(value) = &entry.value {
212                        storage.insert(entry.key.data.clone(), value.data.clone());
213                    }
214                }
215                crate::storage::OpType::Delete => {
216                    storage.remove(&entry.key.data);
217                }
218                crate::storage::OpType::Merge => {
219                    // For now, treat merge as put
220                    if let Some(value) = &entry.value {
221                        storage.insert(entry.key.data.clone(), value.data.clone());
222                    }
223                }
224            }
225        }
226        
227        Ok(())
228    }
229    
230    async fn snapshot(&self) -> Result<Snapshot> {
231        let storage = self.storage.read();
232        let mut snapshot_data = HashMap::new();
233        
234        for (key, value) in storage.iter() {
235            snapshot_data.insert(key.clone(), value.clone());
236        }
237        
238        Ok(Snapshot {
239            data: snapshot_data,
240            timestamp: std::time::SystemTime::now()
241                .duration_since(std::time::UNIX_EPOCH)
242                .unwrap_or_default()
243                .as_millis() as u64,
244        })
245    }
246    
247    async fn close(&self) -> Result<()> {
248        let mut closed = self.closed.write();
249        *closed = true;
250        Ok(())
251    }
252}
