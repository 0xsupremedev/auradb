6/// Breaks a key into the name and label components, with optional default labels.
7///
8/// If any of the default labels are not already present, they will be added to the overall list of labels.
9///
10/// Both the metric name, and labels, are sanitized. See [`sanitize_metric_name`], [`sanitize_label_key`],
11/// and [`sanitize_label_value`] for more information.
12pub fn key_to_parts(
13    key: &Key,
14    default_labels: Option<&IndexMap<String, String>>,
15) -> (String, Vec<String>) {
16    let name = sanitize_metric_name(key.name());
17    let mut values = default_labels.cloned().unwrap_or_default();
18    key.labels().for_each(|label| {
19        values.insert(label.key().to_string(), label.value().to_string());
20    });
21    let labels = values
22        .iter()
23        .map(|(k, v)| format!("{}=\"{}\"", sanitize_label_key(k), sanitize_label_value(v)))
24        .collect();
25
26    (name, labels)
27}

29/// Writes a help (description) line in the Prometheus [exposition format].
30///
31/// [exposition format]: https://github.com/prometheus/docs/blob/main/content/docs/instrumenting/exposition_formats.md#text-format-details
32pub fn write_help_line(buffer: &mut String, name: &str, desc: &str) {
33    buffer.push_str("# HELP ");
34    buffer.push_str(name);
35    buffer.push(' ');
36    let desc = sanitize_description(desc);
37    buffer.push_str(&desc);
38    buffer.push('\n');
39}

52/// Writes a metric in the Prometheus [exposition format].
53///
54/// When `suffix` is specified, it is appended to the `name`, which is useful for writing summary
55/// statistics, such as the sum or total of an aggregated histogram or aggregated summary.  Likewise,
56/// `additional_label` would typically be used to specify a data type-specific label, such as `le` for
57/// for aggregated histograms, or `quantile` for aggregated summaries.
58///
59/// [exposition format]: https://github.com/prometheus/docs/blob/main/content/docs/instrumenting/exposition_formats.md#text-format-details
60pub fn write_metric_line<T, T2>(
61    buffer: &mut String,
62    name: &str,
63    suffix: Option<&'static str>,
64    labels: &[String],
65    additional_label: Option<(&'static str, T)>,
66    value: T2,
67) where
68    T: std::fmt::Display,
69    T2: std::fmt::Display,
70{
71    buffer.push_str(name);
72    if let Some(suffix) = suffix {
73        buffer.push('_');
74        buffer.push_str(suffix);
75    }
76
77    if !labels.is_empty() || additional_label.is_some() {
78        buffer.push('{');
79
80        let mut first = true;
81        for label in labels {
82            if first {
83                first = false;
84            } else {
85                buffer.push(',');
86            }
87            buffer.push_str(label);
88        }
89
90        if let Some((name, value)) = additional_label {
91            if !first {
92                buffer.push(',');
93            }
94            buffer.push_str(name);
95            buffer.push_str("=\"");
96            buffer.push_str(value.to_string().as_str());
97            buffer.push('"');
98        }
99
100        buffer.push('}');
101    }
102
103    buffer.push(' ');
104    buffer.push_str(value.to_string().as_str());
105    buffer.push('\n');
106}

108/// Sanitizes a metric name to be valid under the Prometheus [data model].
109///
110/// [data model]: https://prometheus.io/docs/concepts/data_model/#metric-names-and-labels
111pub fn sanitize_metric_name(name: &str) -> String {
112    // The first character must be [a-zA-Z_:], and all subsequent characters must be [a-zA-Z0-9_:].
113    let mut out = String::with_capacity(name.len());
114    let mut is_invalid: fn(char) -> bool = invalid_metric_name_start_character;
115    for c in name.chars() {
116        if is_invalid(c) {
117            out.push('_');
118        } else {
119            out.push(c);
120        }
121        is_invalid = invalid_metric_name_character;
122    }
123    out
124}

126/// Sanitizes a label key to be valid under the Prometheus [data model].
127///
128/// [data model]: https://prometheus.io/docs/concepts/data_model/#metric-names-and-labels
129pub fn sanitize_label_key(key: &str) -> String {
130    // The first character must be [a-zA-Z_], and all subsequent characters must be [a-zA-Z0-9_].
131    let mut out = String::with_capacity(key.len());
132    let mut is_invalid: fn(char) -> bool = invalid_label_key_start_character;
133    for c in key.chars() {
134        if is_invalid(c) {
135            out.push('_');
136        } else {
137            out.push(c);
138        }
139        is_invalid = invalid_label_key_character;
140    }
141    out
142}

151/// Sanitizes a metric description to be valid under the Prometheus [exposition format].
152///
153/// [exposition format]: https://github.com/prometheus/docs/blob/main/content/docs/instrumenting/exposition_formats.md#text-format-details
154pub fn sanitize_description(value: &str) -> String {
155    sanitize_label_value_or_description(value, true)
156}
157
158fn sanitize_label_value_or_description(value: &str, is_desc: bool) -> String {
159    // All Unicode characters are valid, but backslashes, double quotes, and line feeds must be
160    // escaped.
161    let mut sanitized = String::with_capacity(value.as_bytes().len());
162
163    let mut previous_backslash = false;
164    for c in value.chars() {
165        match c {
166            // Any raw newlines get escaped, period.
167            '\n' => sanitized.push_str("\\n"),
168            // Any double quote we see gets escaped, but only for label values, not descriptions.
169            '"' if !is_desc => {
170                previous_backslash = false;
171                sanitized.push_str("\\\"");
172            }
173            // If we see a backslash, we might be either seeing one that is being used to escape
174            // something, or seeing one that has being escaped. If our last character was a
175            // backslash, then we know this one has already been escaped, and we just emit the
176            // escaped backslash.
177            '\\' => {
178                if previous_backslash {
179                    // This backslash was preceded by another backslash, so we can safely emit an
180                    // escaped backslash.
181                    sanitized.push_str("\\\\");
182                }
183
184                // This may or may not be a backslash that is about to escape something else, so if
185                // we toggle the value here: if it was false, then we're marking ourselves as having
186                // seen a previous backslash (duh) or we just emitted an escaped backslash and now
187                // we're clearing the flag.
188                previous_backslash = !previous_backslash;
189            }
190            c => {
191                // If we had a backslash in holding, and we're here, we know it wasn't escaping
192                // something we care about, so it's on its own, and we emit an escaped backslash,
193                // before emitting the actual character we're handling.
194                if previous_backslash {
195                    previous_backslash = false;
196                    sanitized.push_str("\\\\");
197                }
198                sanitized.push(c);
199            }
200        }
201    }
202
203    // Handle any dangling backslash by writing it out in an escaped fashion.
204    if previous_backslash {
205        sanitized.push_str("\\\\");
206    }
207
208    sanitized
209}

211#[inline]
212fn invalid_metric_name_start_character(c: char) -> bool {
213    // Essentially, needs to match the regex pattern of [a-zA-Z_:].
214    !(c.is_ascii_alphabetic() || c == '_' || c == ':')
215}
216
217#[inline]
218fn invalid_metric_name_character(c: char) -> bool {
219    // Essentially, needs to match the regex pattern of [a-zA-Z0-9_:].
220    !(c.is_ascii_alphanumeric() || c == '_' || c == ':')
221}
222
223#[inline]
224fn invalid_label_key_start_character(c: char) -> bool {
225    // Essentially, needs to match the regex pattern of [a-zA-Z_].
226    !(c.is_ascii_alphabetic() || c == '_')
227}
228
229#[inline]
230fn invalid_label_key_character(c: char) -> bool {
231    // Essentially, needs to match the regex pattern of [a-zA-Z0-9_].
232    !(c.is_ascii_alphanumeric() || c == '_')
233}

360        #[test]
361        fn test_sanitize_label_value(input in "[\n\"\\\\]?.*[\n\"\\\\]?") {
362            let result = sanitize_label_value(&input);
363
364            // If any raw newlines are still present, then we messed up.
365            assert!(!result.contains('\n'), "raw/unescaped newlines present");
366
367            // We specifically remove instances of "\\" because we only care about dangling backslashes.
368            let delayered_backslashes = result.replace("\\\\", "");
369            let as_chars = delayered_backslashes.chars().collect::<Vec<_>>();
370
371            // If the first character is a double quote, then we messed up.
372            assert!(as_chars.first().map(|c| *c != '"').unwrap_or(true),
373                "first character cannot be a double quote: {}", result);
374
375            // Now look for unescaped characters in the rest of the string, in a windowed fashion.
376            let contained_unescaped_chars = as_chars.as_slice()
377                .windows(2)
378                .any(|s| {
379                    let first = s[0];
380                    let second = s[1];
381
382                    match (first, second) {
383                        // If there's a double quote, it has to have been preceded by an escaping
384                        // backslash.
385                        (c, '"') => c != '\\',
386                        // If there's a backslash, it can only be in front of an 'n' for escaping
387                        // newlines.
388                        ('\\', c) => c != 'n',
389                        // Everything else is valid.
390                        _ => false,
391                    }
392                });
393            assert!(!contained_unescaped_chars, "invalid or missing escape detected");
394        }
