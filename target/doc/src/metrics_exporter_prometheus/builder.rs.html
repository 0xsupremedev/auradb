58enum ExporterConfig {
59    // Run an HTTP listener on the given `listen_address`.
60    #[cfg(feature = "http-listener")]
61    HttpListener { listen_address: SocketAddr },
62
63    // Run a push gateway task sending to the given `endpoint` after `interval` time has elapsed,
64    // infinitely.
65    #[cfg(feature = "push-gateway")]
66    PushGateway {
67        endpoint: Uri,
68        interval: Duration,
69        username: Option<String>,
70        password: Option<String>,
71    },
72
73    #[allow(dead_code)]
74    Unconfigured,
75}

77impl ExporterConfig {
78    #[cfg_attr(not(any(feature = "http-listener", feature = "push-gateway")), allow(dead_code))]
79    fn as_type_str(&self) -> &'static str {
80        match self {
81            #[cfg(feature = "http-listener")]
82            Self::HttpListener { .. } => "http-listener",
83            #[cfg(feature = "push-gateway")]
84            Self::PushGateway { .. } => "push-gateway",
85            Self::Unconfigured => "unconfigured,",
86        }
87    }

183    /// Adds an IP address or subnet to the allowlist for the scrape endpoint.
184    ///
185    /// If a client makes a request to the scrape endpoint and their IP is not present in the
186    /// allowlist, either directly or within any of the allowed subnets, they will receive a 403
187    /// Forbidden response.
188    ///
189    /// Defaults to allowing all IPs.
190    ///
191    /// ## Security Considerations
192    ///
193    /// On its own, an IP allowlist is insufficient for access control, if the exporter is running
194    /// in an environment alongside applications (such as web browsers) that are susceptible to [DNS
195    /// rebinding](https://en.wikipedia.org/wiki/DNS_rebinding) attacks.
196    ///
197    /// ## Errors
198    ///
199    /// If the given address cannot be parsed into an IP address or subnet, an error variant will be
200    /// returned describing the error.
201    #[cfg(feature = "http-listener")]
202    #[cfg_attr(docsrs, doc(cfg(feature = "http-listener")))]
203    pub fn add_allowed_address<A>(mut self, address: A) -> Result<Self, BuildError>
204    where
205        A: AsRef<str>,
206    {
207        use std::str::FromStr;
208
209        let address = IpNet::from_str(address.as_ref())
210            .map_err(|e| BuildError::InvalidAllowlistAddress(e.to_string()))?;
211        self.allowed_addresses.get_or_insert(vec![]).push(address);
212
213        Ok(self)
214    }

216    /// Sets the quantiles to use when rendering histograms.
217    ///
218    /// Quantiles represent a scale of 0 to 1, where percentiles represent a scale of 1 to 100, so
219    /// a quantile of 0.99 is the 99th percentile, and a quantile of 0.99 is the 99.9th percentile.
220    ///
221    /// Defaults to a hard-coded set of quantiles: 0.0, 0.5, 0.9, 0.95, 0.99, 0.999, and 1.0. This means
222    /// that all histograms will be exposed as Prometheus summaries.
223    ///
224    /// If buckets are set (via [`set_buckets`][Self::set_buckets] or
225    /// [`set_buckets_for_metric`][Self::set_buckets_for_metric]) then all histograms will be exposed
226    /// as summaries instead.
227    ///
228    /// ## Errors
229    ///
230    /// If `quantiles` is empty, an error variant will be thrown.
231    pub fn set_quantiles(mut self, quantiles: &[f64]) -> Result<Self, BuildError> {
232        if quantiles.is_empty() {
233            return Err(BuildError::EmptyBucketsOrQuantiles);
234        }
235
236        self.quantiles = parse_quantiles(quantiles);
237        Ok(self)
238    }

240    /// Sets the buckets to use when rendering histograms.
241    ///
242    /// Buckets values represent the higher bound of each buckets.  If buckets are set, then all
243    /// histograms will be rendered as true Prometheus histograms, instead of summaries.
244    ///
245    /// ## Errors
246    ///
247    /// If `values` is empty, an error variant will be thrown.
248    pub fn set_buckets(mut self, values: &[f64]) -> Result<Self, BuildError> {
249        if values.is_empty() {
250            return Err(BuildError::EmptyBucketsOrQuantiles);
251        }
252
253        self.buckets = Some(values.to_vec());
254        Ok(self)
255    }

257    /// Sets the bucket for a specific pattern.
258    ///
259    /// The match pattern can be a full match (equality), prefix match, or suffix match.  The
260    /// matchers are applied in that order if two or more matchers would apply to a single metric.
261    /// That is to say, if a full match and a prefix match applied to a metric, the full match would
262    /// win, and if a prefix match and a suffix match applied to a metric, the prefix match would win.
263    ///
264    /// Buckets values represent the higher bound of each buckets.  If buckets are set, then any
265    /// histograms that match will be rendered as true Prometheus histograms, instead of summaries.
266    ///
267    /// This option changes the observer's output of histogram-type metric into summaries.
268    /// It only affects matching metrics if [`set_buckets`][Self::set_buckets] was not used.
269    ///
270    /// ## Errors
271    ///
272    /// If `values` is empty, an error variant will be thrown.
273    pub fn set_buckets_for_metric(
274        mut self,
275        matcher: Matcher,
276        values: &[f64],
277    ) -> Result<Self, BuildError> {
278        if values.is_empty() {
279            return Err(BuildError::EmptyBucketsOrQuantiles);
280        }
281
282        let buckets = self.bucket_overrides.get_or_insert_with(HashMap::new);
283        buckets.insert(matcher.sanitized(), values.to_vec());
284        Ok(self)
285    }

287    /// Sets the idle timeout for metrics.
288    ///
289    /// If a metric hasn't been updated within this timeout, it will be removed from the registry
290    /// and in turn removed from the normal scrape output until the metric is emitted again.  This
291    /// behavior is driven by requests to generate rendered output, and so metrics will not be
292    /// removed unless a request has been made recently enough to prune the idle metrics.
293    ///
294    /// Further, the metric kind "mask" configures which metrics will be considered by the idle
295    /// timeout.  If the kind of a metric being considered for idle timeout is not of a kind
296    /// represented by the mask, it will not be affected, even if it would have othered been removed
297    /// for exceeding the idle timeout.
298    ///
299    /// Refer to the documentation for [`MetricKindMask`](metrics_util::MetricKindMask) for more
300    /// information on defining a metric kind mask.
301    #[must_use]
302    pub fn idle_timeout(mut self, mask: MetricKindMask, timeout: Option<Duration>) -> Self {
303        self.idle_timeout = timeout;
304        self.recency_mask = if self.idle_timeout.is_none() { MetricKindMask::NONE } else { mask };
305        self
306    }

309    ///
310    /// Global labels are applied to all metrics.  Labels defined on the metric key itself have precedence
311    /// over any global labels.  If this method is called multiple times, the latest value for a given label
312    /// key will be used.

315    where
316        K: Into<String>,
317        V: Into<String>,

319        let labels = self.global_labels.get_or_insert_with(IndexMap::new);

324    /// Builds the recorder and exporter and installs them globally.
325    ///
326    /// When called from within a Tokio runtime, the exporter future is spawned directly
327    /// into the runtime.  Otherwise, a new single-threaded Tokio runtime is created
328    /// on a background thread, and the exporter is spawned there.
329    ///
330    /// ## Errors
331    ///
332    /// If there is an error while either building the recorder and exporter, or installing the
333    /// recorder and exporter, an error variant will be returned describing the error.
334    #[cfg(any(feature = "http-listener", feature = "push-gateway"))]
335    #[cfg_attr(docsrs, doc(cfg(any(feature = "http-listener", feature = "push-gateway"))))]
336    pub fn install(self) -> Result<(), BuildError> {
337        let recorder = if let Ok(handle) = runtime::Handle::try_current() {
338            let (recorder, exporter) = {
339                let _g = handle.enter();
340                self.build()?
341            };
342
343            handle.spawn(exporter);
344
345            recorder
346        } else {
347            let thread_name =
348                format!("metrics-exporter-prometheus-{}", self.exporter_config.as_type_str());
349
350            let runtime = runtime::Builder::new_current_thread()
351                .enable_all()
352                .build()
353                .map_err(|e| BuildError::FailedToCreateRuntime(e.to_string()))?;
354
355            let (recorder, exporter) = {
356                let _g = runtime.enter();
357                self.build()?
358            };
359
360            thread::Builder::new()
361                .name(thread_name)
362                .spawn(move || runtime.block_on(exporter))
363                .map_err(|e| BuildError::FailedToCreateRuntime(e.to_string()))?;
364
365            recorder
366        };
367
368        metrics::set_boxed_recorder(Box::new(recorder))?;
369
370        Ok(())
371    }

390    /// Builds the recorder and exporter and returns them both.
391    ///
392    /// In most cases, users should prefer to use [`install`][PrometheusBuilder::install] to create
393    /// and install the recorder and exporter automatically for them.  If a caller is combining
394    /// recorders, or needs to schedule the exporter to run in a particular way, this method, or
395    /// [`build_recorder`][PrometheusBuilder::build_recorder], provide the flexibility to do so.
396    ///
397    /// ## Panics
398    ///
399    /// This method must be called from within an existing Tokio runtime or it will panic.
400    ///
401    /// ## Errors
402    ///
403    /// If there is an error while building the recorder and exporter, an error variant will be
404    /// returned describing the error.
405    #[cfg(any(feature = "http-listener", feature = "push-gateway"))]
406    #[cfg_attr(docsrs, doc(cfg(any(feature = "http-listener", feature = "push-gateway"))))]
407    #[cfg_attr(not(feature = "http-listener"), allow(unused_mut))]
408    pub fn build(mut self) -> Result<(PrometheusRecorder, ExporterFuture), BuildError> {
409        #[cfg(feature = "http-listener")]
410        let allowed_addresses = self.allowed_addresses.take();
411
412        let exporter_config = self.exporter_config.clone();
413        let recorder = self.build_recorder();
414        let handle = recorder.handle();
415
416        match exporter_config {
417            ExporterConfig::Unconfigured => Err(BuildError::MissingExporterConfiguration),
418            #[cfg(feature = "http-listener")]
419            ExporterConfig::HttpListener { listen_address } => {
420                let server = Server::try_bind(&listen_address)
421                    .map_err(|e| BuildError::FailedToCreateHTTPListener(e.to_string()))?;
422                let exporter = async move {
423                    let make_svc = make_service_fn(move |socket: &AddrStream| {
424                        let remote_addr = socket.remote_addr().ip();
425
426                        // If the allowlist is empty, the request is allowed.  Otherwise, it must
427                        // match one of the entries in the allowlist or it will be denied.
428                        let is_allowed = allowed_addresses.as_ref().map_or(true, |addresses| {
429                            addresses.iter().any(|address| address.contains(&remote_addr))
430                        });
431
432                        let handle = handle.clone();
433
434                        async move {
435                            Ok::<_, hyper::Error>(service_fn(move |_| {
436                                let handle = handle.clone();
437
438                                async move {
439                                    if is_allowed {
440                                        let output = handle.render();
441                                        Ok::<_, hyper::Error>(Response::new(Body::from(output)))
442                                    } else {
443                                        Ok::<_, hyper::Error>(
444                                            Response::builder()
445                                                .status(StatusCode::FORBIDDEN)
446                                                .body(Body::empty())
447                                                .expect("static response is valid"),
448                                        )
449                                    }
450                                }
451                            }))
452                        }
453                    });
454                    server.serve(make_svc).await
455                };
456
457                Ok((recorder, Box::pin(exporter)))
458            }
459
460            #[cfg(feature = "push-gateway")]
461            ExporterConfig::PushGateway { endpoint, interval, username, password } => {
462                let exporter = async move {
463                    let client = Client::new();
464                    let auth = username
465                        .as_ref()
466                        .map(|name| basic_auth(name, password.as_ref().map(|x| &**x)));
467
468                    loop {
469                        // Sleep for `interval` amount of time, and then do a push.
470                        tokio::time::sleep(interval).await;
471
472                        let mut builder = Request::builder();
473                        if let Some(auth) = &auth {
474                            builder = builder.header("authorization", auth.clone());
475                        }
476
477                        let output = handle.render();
478                        let result = builder
479                            .method(Method::PUT)
480                            .uri(endpoint.clone())
481                            .body(Body::from(output));
482                        let req = match result {
483                            Ok(req) => req,
484                            Err(e) => {
485                                error!("failed to build push gateway request: {}", e);
486                                continue;
487                            }
488                        };
489
490                        match client.request(req).await {
491                            Ok(response) => {
492                                if !response.status().is_success() {
493                                    let status = response.status();
494                                    let status = status
495                                        .canonical_reason()
496                                        .unwrap_or_else(|| status.as_str());
497                                    let body = aggregate(response.into_body()).await;
498                                    let body = body
499                                        .map_err(|_| ())
500                                        .map(|mut b| b.copy_to_bytes(b.remaining()))
501                                        .map(|b| b[..].to_vec())
502                                        .and_then(|s| String::from_utf8(s).map_err(|_| ()))
503                                        .unwrap_or_else(|_| {
504                                            String::from("<failed to read response body>")
505                                        });
506                                    error!(
507                                        message = "unexpected status after pushing metrics to push gateway",
508                                        status,
509                                        %body,
510                                    );
511                                }
512                            }
513                            Err(e) => error!("error sending request to push gateway: {:?}", e),
514                        }
515                    }
516                };
517
518                Ok((recorder, Box::pin(exporter)))
519            }
520        }
521    }

523    /// Builds the recorder and returns it.
524    pub fn build_recorder(self) -> PrometheusRecorder {
525        self.build_with_clock(Clock::new())
526    }

528    pub(crate) fn build_with_clock(self, clock: Clock) -> PrometheusRecorder {
529        let inner = Inner {
530            registry: Registry::new(GenerationalStorage::new(AtomicStorage)),
531            recency: Recency::new(clock, self.recency_mask, self.idle_timeout),
532            distributions: RwLock::new(HashMap::new()),
533            distribution_builder: DistributionBuilder::new(
534                self.quantiles,
535                self.buckets,
536                self.bucket_overrides,
537            ),
538            descriptions: RwLock::new(HashMap::new()),
539            global_labels: self.global_labels.unwrap_or_default(),
540        };
541
542        PrometheusRecorder::from(inner)
543    }
544}

552#[cfg(feature = "push-gateway")]
553fn basic_auth(username: &str, password: Option<&str>) -> HeaderValue {
554    use base64::prelude::BASE64_STANDARD;
555    use base64::write::EncoderWriter;
556    use std::io::Write;
557
558    let mut buf = b"Basic ".to_vec();
559    {
560        let mut encoder = EncoderWriter::new(&mut buf, &BASE64_STANDARD);
561        let _ = write!(encoder, "{username}:");
562        if let Some(password) = password {
563            let _ = write!(encoder, "{password}");
564        }
565    }
566    let mut header = HeaderValue::from_bytes(&buf).expect("base64 is always valid HeaderValue");
567    header.set_sensitive(true);
568    header
569}

942    fn test_idle_timeout_catches_delayed_idle() {
943        let (clock, mock) = Clock::mock();
944
945        let recorder = PrometheusBuilder::new()
946            .idle_timeout(MetricKindMask::ALL, Some(Duration::from_secs(10)))
947            .build_with_clock(clock);
948
949        let key = Key::from_name("basic_counter");
950        let counter1 = recorder.register_counter(&key);
951        counter1.increment(42);
952
953        // First render, which starts tracking the counter in the recency state.
954        let handle = recorder.handle();
955        let rendered = handle.render();
956        let expected = concat!("# TYPE basic_counter counter\n", "basic_counter 42\n\n",);
957
958        assert_eq!(rendered, expected);
959
960        // Now go forward by 9 seconds, which is close but still right unfer the idle timeout.
961        mock.increment(Duration::from_secs(9));
962        let rendered = handle.render();
963        assert_eq!(rendered, expected);
964
965        // Now increment the counter and advance time by two seconds: this pushes it over the idle
966        // timeout threshold, but it should not be removed since it has been updated.
967        counter1.increment(1);
968
969        let expected_after = concat!("# TYPE basic_counter counter\n", "basic_counter 43\n\n",);
970
971        mock.increment(Duration::from_secs(2));
972        let rendered = handle.render();
973        assert_eq!(rendered, expected_after);
974
975        // Now advance by 11 seconds, right past the idle timeout threshold.  We've made no further
976        // updates to the counter so it should be properly removed this time.
977        mock.increment(Duration::from_secs(11));
978        let rendered = handle.render();
979        assert_eq!(rendered, "");
980    }
