13    /// The recorder is uninitialized.
14    const UNINITIALIZED: usize = 0;
15
16    /// The recorder is currently being initialized.
17    const INITIALIZING: usize = 1;
18
19    /// The recorder has been initialized successfully and can be read.
20    const INITIALIZED: usize = 2;
21
22    /// An specialized version of `OnceCell` for `Recorder`.
23    pub struct RecorderOnceCell {
24        recorder: UnsafeCell<Option<&'static dyn Recorder>>,
25        state: AtomicUsize,
26    }
27
28    impl RecorderOnceCell {
29        /// Creates an uninitialized `RecorderOnceCell`.
30        pub const fn new() -> Self {
31            Self { recorder: UnsafeCell::new(None), state: AtomicUsize::new(UNINITIALIZED) }
32        }
33
34        pub fn set(&self, recorder: &'static dyn Recorder) -> Result<(), SetRecorderError> {
35            // Try and transition the cell from `UNINITIALIZED` to `INITIALIZING`, which would give
36            // us exclusive access to set the recorder.
37            match self.state.compare_exchange(
38                UNINITIALIZED,
39                INITIALIZING,
40                Ordering::Acquire,
41                Ordering::Relaxed,
42            ) {
43                Ok(UNINITIALIZED) => {
44                    unsafe {
45                        // SAFETY: Access is unique because we can only be here if we won the race
46                        // to transition from `UNINITIALIZED` to `INITIALIZING` above.
47                        self.recorder.get().write(Some(recorder));
48                    }
49
50                    // Mark the recorder as initialized, which will make it visible to readers.
51                    self.state.store(INITIALIZED, Ordering::Release);
52                    Ok(())
53                }
54                _ => Err(SetRecorderError(())),
55            }
56        }
57
58        /// Clears the currently installed recorder, allowing a new writer to override it.
59        ///
60        /// # Safety
61        ///
62        /// The caller must guarantee that no reader has read the state before we do this and then
63        /// reads the recorder after another writer has written to it after us.
64        pub unsafe fn clear(&self) {
65            // Set the state to `UNINITIALIZED` to allow the next writer to write again. This is not
66            // a problem for readers since their `&'static` refs will remain valid forever.
67            self.state.store(UNINITIALIZED, Ordering::Relaxed);
68        }
69
70        pub fn try_load(&self) -> Option<&'static dyn Recorder> {
71            if self.state.load(Ordering::Acquire) != INITIALIZED {
72                None
73            } else {
74                // SAFETY: If the state is `INITIALIZED`, then we know that the recorder has been
75                // installed and is safe to read.
76                unsafe { self.recorder.get().read() }
77            }
78        }
79    }
80
81    // SAFETY: We can only mutate through `set`, which is protected by the `state` and unsafe
82    // function where the caller has to guarantee synced-ness.
83    unsafe impl Send for RecorderOnceCell {}
84    unsafe impl Sync for RecorderOnceCell {}
85}

92/// A trait for registering and recording metrics.
93///
94/// This is the core trait that allows interoperability between exporter implementations and the
95/// macros provided by `metrics`.
96pub trait Recorder {
97    /// Describes a counter.
98    ///
99    /// Callers may provide the unit or a description of the counter being registered. Whether or
100    /// not a metric can be reregistered to provide a unit/description, if one was already passed
101    /// or not, as well as how units/descriptions are used by the underlying recorder, is an
102    /// implementation detail.
103    fn describe_counter(&self, key: KeyName, unit: Option<Unit>, description: SharedString);
104
105    /// Describes a gauge.
106    ///
107    /// Callers may provide the unit or a description of the gauge being registered. Whether or
108    /// not a metric can be reregistered to provide a unit/description, if one was already passed
109    /// or not, as well as how units/descriptions are used by the underlying recorder, is an
110    /// implementation detail.
111    fn describe_gauge(&self, key: KeyName, unit: Option<Unit>, description: SharedString);
112
113    /// Describes a histogram.
114    ///
115    /// Callers may provide the unit or a description of the histogram being registered. Whether or
116    /// not a metric can be reregistered to provide a unit/description, if one was already passed
117    /// or not, as well as how units/descriptions are used by the underlying recorder, is an
118    /// implementation detail.
119    fn describe_histogram(&self, key: KeyName, unit: Option<Unit>, description: SharedString);
120
121    /// Registers a counter.
122    fn register_counter(&self, key: &Key) -> Counter;
123
124    /// Registers a gauge.
125    fn register_gauge(&self, key: &Key) -> Gauge;
126
127    /// Registers a histogram.
128    fn register_histogram(&self, key: &Key) -> Histogram;
129}

152/// Sets the global recorder to a `&'static Recorder`.
153///
154/// This function may only be called once in the lifetime of a program.  Any metrics recorded
155/// before the call to `set_recorder` occurs will be completely ignored.
156///
157/// This function does not typically need to be called manually.  Metrics implementations should
158/// provide an initialization method that installs the recorder internally.
159///
160/// # Errors
161///
162/// An error is returned if a recorder has already been set.
163pub fn set_recorder(recorder: &'static dyn Recorder) -> Result<(), SetRecorderError> {
164    RECORDER.set(recorder)
165}
166
167/// Sets the global recorder to a `Box<Recorder>`.
168///
169/// This is a simple convenience wrapper over `set_recorder`, which takes a `Box<Recorder>`
170/// rather than a `&'static Recorder`.  See the documentation for [`set_recorder`] for more
171/// details.
172///
173/// # Errors
174///
175/// An error is returned if a recorder has already been set.
176pub fn set_boxed_recorder(recorder: Box<dyn Recorder>) -> Result<(), SetRecorderError> {
177    RECORDER.set(Box::leak(recorder))
178}

180/// Clears the currently configured recorder.
181///
182/// This will leak the currently installed recorder, as we cannot safely drop it due to it being
183/// provided via a reference with a `'static` lifetime.
184///
185/// This method is typically only useful for testing or benchmarking.
186///
187/// # Safety
188///
189/// The caller must ensure that this method is not being called while other threads are either
190/// loading a reference to the global recorder, or attempting to initialize the global recorder, as
191/// it can cause a data race.
192#[doc(hidden)]
193pub unsafe fn clear_recorder() {
194    RECORDER.clear();
195}
